### WebGPU Availability
Not all browsers support WebGPU yet. Fallback strategy:

```rust
pub async fn create_embedder() -> Result<Box<dyn Embedder>> {
    // Try WebGPU first
    if let Ok(wgpu_embedder) = EmbeddingGenerator::<Wgpu>::new().await {
        return Ok(Box::new(wgpu_embedder));
    }
    
    // Fall back to CPU (slower but works everywhere)
    if let Ok(cpu_embedder) = EmbeddingGenerator::<NdArray>::new().await {
        return Ok(Box::new(cpu_embedder));
    }
    
    // Last resort: hash-based (Phase 5 fallback)
    Ok(Box::new(HashEmbedder::new()))
}
```
