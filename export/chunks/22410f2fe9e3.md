//! Utility functions

/// Process a vector of data.
pub fn process_data(data: Vec<i32>) -> Vec<i32> {
    data.iter().map(|x| x * 2).collect()
}

/// Calculate the sum of a vector.
pub fn sum(data: &[i32]) -> i32 {
    data.iter().sum()
}
