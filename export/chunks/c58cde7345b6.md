function isAllowedPath(path) {
  const lower = path.toLowerCase();
  if (SKIP_DIRS.some((dir) => lower.includes(`/${dir}/`) || lower.startsWith(`${dir}/`))) {
    return false;
  }
  return ALLOWED_EXTENSIONS.some((ext) => lower.endsWith(ext));
}
