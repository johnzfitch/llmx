function scheduleSearch() {
  if (!state.indexLoaded || state.busy) {
    return;
  }
  if (searchTimer) {
    clearTimeout(searchTimer);
  }
  searchTimer = setTimeout(async () => {
    searchTimer = null;
    await runSearch();
  }, 200);
}
