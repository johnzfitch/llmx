function formatErrorForUi(error) {
  const message = error instanceof Error ? error.message : String(error || "Unknown error");
  const cleaned = message.replace(/\s+/g, " ").trim();
  return cleaned.length > 240 ? `${cleaned.slice(0, 237)}...` : cleaned;
}
